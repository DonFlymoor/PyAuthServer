[docs]    def server_execute(self, command: TypeFlag(str, max_length=1000)) -> Netmodes.server:
        with stdout_io() as s:
            try:
                # Create a code object in order to print result
                code = compile(command, "<dummy>", "single")
                exec(code, self.data)

            except Exception:
                print_exc()

        result = s.getvalue()[:-1]
        self.result(result)
