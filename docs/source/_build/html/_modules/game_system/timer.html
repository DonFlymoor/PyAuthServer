[docs]    def reset(self):
        """Reset timer to start value"""
        self.value = self.start
        self.active = True

        if callable(self.on_reset):
            self.on_reset()



[docs]    def stop(self):
        """Stop timer updating"""
        self.value = self.end
        self.active = False

        if callable(self.on_stop):
            self.on_stop()

        if self.disposable:
            self.delete()



[docs]    def update(self, delta_time):
        """Update timer value

        :param delta_time: time since last update
        """
        if not self.active:
            return

        self.value = self.update_operator(self.value, delta_time)

        if callable(self.on_update):
            self.on_update()

        if self.comparison_operator(self.value, self.end):

            if callable(self.on_target):
                self.on_target()

            if self.repeat:
                self.reset()

            else:
                self.stop()
