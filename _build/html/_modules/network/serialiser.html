[docs]def handler_from_bit_length(bits):
    bytes_length = bits2bytes(bits)

    last_packer = None
    for packer in int_packers:
        if packer.size() > bytes_length:
            break

        last_packer = packer

    else:
        raise ValueError("Integer too large to pack")

    return last_packer
